class Solution {
public:

    unordered_map<int, vector<int>> um;
    
    Solution(vector<int>& nums) {
        for (int i = 0; i < nums.size(); i ++) {
            um[nums[i]].push_back(i);
        }
    }
    
    int pick(int target) {
        return um[target][rand() % um[target].size()];
    }
};
